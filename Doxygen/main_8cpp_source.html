36void setHUPHandler() {
37 struct sigaction act;
38 memset (&act, 0, sizeof (act));
39 act.sa_handler = sigHandler;
40 if (sigaction (SIGHUP, &act, NULL) < 0) {
41 perror ("sigaction");
